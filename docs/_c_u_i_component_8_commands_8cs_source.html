44 /// <summary>
45 /// Methods ending in "Command" will be added as commands
46 /// </summary>
47 public virtual void AddCommands()
48 {
49 foreach (MethodInfo mi in this.GetType().GetMethods())
50 {
51 if (Attribute.IsDefined(mi, typeof(CommandAttribute)))
52 {
53 try
54 {
55 string name = mi.Name;
56 if (name != "Command" && name.EndsWith("Command"))
57 {
58 name = name.Substring(0, name.Length - "Command".Length);
59 }
60 AddCommad(name, mi.CreateDelegate<Action>(this));
61 }
62 catch (Exception e)
63 {
64 Info($"{e.Message}\nMethod: {this.GetType()}.{mi.Name}");
65 }
66 }
67 }
68 }
