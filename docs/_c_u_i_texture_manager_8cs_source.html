92 public Texture2D GetTexture(string path)
93 {
94 if (LoadedTextures.ContainsKey(path)) return LoadedTextures[path];
95
96 Texture2D loaded = null;
97
98 if (CUI.AssetsPath != null) loaded ??= TryLoad(Path.Combine(CUI.AssetsPath, path));
99 if (PGNAssets != null) loaded ??= TryLoad(Path.Combine(PGNAssets, path));
100 loaded ??= TryLoad(path);
101 if (loaded == null)
102 {
103 CUI.Warning($"Coudn't find {path} texture, setting it to backup texture");
104 loaded ??= BackupTexture;
105 if (PGNAssets == null)
106 {
107 CUI.Warning($"Note: It was loaded before CUI.PGNAssets was set");
108 }
109 }
110
111 LoadedTextures[path] = loaded;
112 return loaded;
113 }
