185 public override void FromXML(XElement element)
186 {
187 foreach (XElement childElement in element.Element("Children").Elements())
188 {
189 Type childType = CUIReflection.GetComponentTypeByName(childElement.Name.ToString());
190 if (childType == null) continue;
191
192 CUIComponent child = (CUIComponent)Activator.CreateInstance(childType);
193 child.FromXML(childElement);
194
195 this.Append(child, child.AKA);
196 }
197
198 foreach (XElement link in element.Element("Connections").Elements())
199 {
200 CUIComponent startComponent = this[link.Attribute("Start").Value];
201 CUIComponent endComponent = this[link.Attribute("End").Value];
202
203 if (startComponent == null || endComponent == null)
204 {
205 CUIDebug.Error("startComponent == null || endComponent == null");
206 continue;
207 }
208 Connect(link.Attribute("Start").Value, link.Attribute("End").Value);
209 }
210
211 //TODO: think, this is potentially very bugged,
212 // Some props might need to be assigned before children, and some after
213 ExtractProps(element);
